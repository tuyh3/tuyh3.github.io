nc命令批量测试端口脚本

nc命令测试端口：

cat /dev/null | nc <hostname> <port>

批量测试脚本：

#!/bin/bash
string="10.72.228.21;10.72.236.24;10.70.186.21;10.76.232.20;10.78.198.230"

array=(${string//;/ })
for var in ${array[@]}
do
echo "=============test ip: $var link start==================="
touch "logs/$var.log"
cat /dev/null| nc -v -w 5  -t $var $1
if [ $? -eq 0 ];then
    echo "$var:$1    connect success" >>"logs/$var.log"
else
    echo "$var:$1    connect failed" >>"logs/$var.log"
fi

echo "=============test ip: $var link end==================="
done
